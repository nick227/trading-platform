model bot {
  id          String       @id
  userId      String
  portfolioId String
  strategyId  String?
  templateId  String?

  enabled     Boolean      @default(true)
  config      Json
  createdAt   DateTime     @default(now())
  updatedAt   DateTime
  deletedAt   DateTime?
  portfolio   portfolio    @relation(fields: [portfolioId], references: [id], map: "Bot_portfolioId_fkey")
  strategy    strategy?    @relation(fields: [strategyId], references: [id], map: "Bot_strategyId_fkey")
  bottemplate bottemplate? @relation(fields: [templateId], references: [id], map: "Bot_templateId_fkey")

  botrule     botrule[]
  execution   execution[]

  @@index([enabled], map: "Bot_enabled_idx")
  @@index([portfolioId], map: "Bot_portfolioId_idx")
  @@index([strategyId], map: "Bot_strategyId_fkey")
  @@index([templateId], map: "Bot_templateId_idx")
  @@index([userId], map: "Bot_userId_idx")
}
